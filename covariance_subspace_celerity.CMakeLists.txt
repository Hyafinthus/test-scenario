cmake_minimum_required(VERSION 3.13)
project(covariance_subspace_celerity LANGUAGES CXX)

find_package(Celerity 0.6.0 REQUIRED)
add_executable(covariance_subspace_celerity
               covariance_subspace_celerity.cpp)
set_property(TARGET covariance_subspace_celerity
             PROPERTY CXX_STANDARD ${CELERITY_CXX_STANDARD})
add_celerity_to_target(
  TARGET covariance_subspace_celerity
  SOURCES covariance_subspace_celerity.cpp)
